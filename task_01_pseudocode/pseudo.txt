Algorithm 1:
 Largest number until the user enters 0
set largest_number to 0
request input from the user: "Enter a positive number, enter 0 to stop"
store input in variable called number
while number is not equal to 0
    if number is greater than largest_number
        set largest_number to number
    request input from the user: "Enter a positive number, enter 0 to stop"
    store input in variable called number
print "Largest number entered is"
print largest_number

Algorithm 4: Grocery list, print items left to buy in alphabetical order
set items_left_to_buy to an empty list
request input from the user: "Enter any of the item you still need to buy, or type 'done' to finish"
store input in variable called item
while item is not equal to "done"
    add item to items_left_to_buy
    request input from the user: "Enter an item you still need to buy, or type 'done' to finish"
    store input in variable called item
sort items_left_to_buy in alphabetical order
print "Items left to buy, in alphabetical order"
for each item in items_left_to_buy
    print item
